选择器可以分组

CSS，子元素从父元素继承属性

        /* 分组 */
        h1,h2,h3,h4,h5,h6 {
                color: green;
        }  
    
        /* 选中 <p></p> */
        p{color: red;}

元素选择器

        /*标签选择器*/
        p{ }
        

        <body>
            <p>标签选择器</p>
        </body>
    
        /*id选择器*/
        #red
        

        <body>
            <p id="red">id选择器</p>
        </body>
    
        /*class选择器*/
        .green
        

        <body>
            <p class="green">class选择器</p>
        </body>
    
        /*属性选择器*/
        [title=mytitle]{}
        

        <body>
            <p title="mytitle"> </p>
        </body>
优先级: id>class>标签

派生选择器contextual selectors

        <div>
            <span></span>
            <p>
                <span></span>
            </p>
        </div>
    
        /* 选中包含的元素 */
        div span{}
        /* 选中情况 */
        <div>
            <span></span>
            <p>
                <span></span>
            </p>
        </div>
    
        /* 选中包含的元素 */
        div>span{}
        /* 选中情况 */
        <div>
            <span></span>
            <p>
                <span></span>
            </p>
        </div>
